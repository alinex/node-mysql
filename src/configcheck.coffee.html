      title: "Hostname"
      description: "the hostname or ip address to connect to"
      type: 'hostname'
      default: 'localhost'
    port:
      title: "Port"
      description: "the port mysql is listening"
      type: 'integer'

    database:
      title: "Database Name"

      type: 'string'

    user:
      title: "Username"

      type: 'string'
    password:
      title: "Password"
      description: "the password to login"
      type: 'string'
      optional: true
    charset:
      title: "Default Charset"

      type: 'string'

      type: 'string'
      default: 'local'

      unit: 'ms'
